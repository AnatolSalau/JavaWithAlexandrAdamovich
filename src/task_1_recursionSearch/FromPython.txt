all_data = {
    'a': {
        'b': {
            'c': ''
        }
    },
    'd': '',
    'f': {
        'g': '',
        'h': {
            'j': ''
        }
    }
}


all_worker = [] # сюда записываем всех рабочих
queue = [] # это очередь данных для обработки данных

# функция которая получает очередь, и создат новую
# паралельно добавляя работников в список
def parse_collection(data):
    global all_worker # питоновская проблема, нужно добавить переменные
    global queue # из вне функции
    old_q = data.copy() # копируем очередь для работы с ней
    queue = {} # удаляем старую очередь. Будем тут создавать новую
    for k, v in old_q.items(): # цикл по словарю k ключ, v значение
        if isinstance(v, dict): # проверка есть ли вложенные данные
            queue.update(v) # если есть, добавляем в новую очередь
            all_worker += k # добавляем работника
        else:
            all_worker += k # добавляем работника

queue = all_data # добавляем данные в очередь

while True:
    if queue: # если есть очередь
        parse_collection(queue) # парсим
    else: # иначе
        break # останавливаем цикл


print(all_worker)
